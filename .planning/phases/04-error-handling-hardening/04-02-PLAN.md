### ERR-02: Failure Classification

1. **Modify `#processToken`** — Replace the single catch with structured error handling. Use a stage marker to classify failures without parsing error messages (per Pitfall 4):
   ```javascript
   static async #processToken(tokenDoc, index, processedIds) {
     const creatureName = tokenDoc.actor?.name || tokenDoc.name;

     if (processedIds.has(tokenDoc.id)) {
       Logger.log(`Skipping already processed token: ${tokenDoc.name}`);
       return { status: "skipped", name: creatureName };
     }

     if (!canvas.scene.tokens.has(tokenDoc.id)) {
       Logger.log(`Token "${tokenDoc.name}" no longer exists, skipping`);
       return { status: "skipped", name: creatureName };
     }

     processedIds.add(tokenDoc.id);

     const match = NameMatcher.findMatch(creatureName, index);
     if (!match) {
       return { status: "not_found", name: creatureName };
     }

     // Structured error classification: wrap import and creation stages separately
     let replaceResult;
     try {
       replaceResult = await TokenReplacer.replaceToken(tokenDoc, match.entry, match.pack);
     } catch (error) {
       // Classify based on what replaceToken was doing when it failed.
       // replaceToken throws from: getDocument (import), #getOrImportWorldActor (import),
       // createEmbeddedDocuments (creation), or deleteEmbeddedDocuments (creation).
       // Use error message heuristic: if error mentions "import" → import_failed, else creation_failed.
       // Better approach: Add a `stage` property to errors thrown from replaceToken.
       //
       // SIMPLEST approach (per research Pitfall 4): check if the error comes from the import stage
       // by looking for "import" in the error message. However, this is fragile.
       //
       // RECOMMENDED approach: Wrap the two stages in #processToken with a stage variable:
       const isImportError = error.message?.toLowerCase().includes("import") ||
                             error.message?.toLowerCase().includes("failed to load") ||
                             error.message?.toLowerCase().includes("getdocument");
       const status = isImportError ? "import_failed" : "creation_failed";
       Logger.error(`Error replacing token ${tokenDoc.name} (${status})`, error);
       return { status, name: creatureName };
     }

     return { status: "replaced", name: creatureName };
   }
   ```

   **ALTERNATIVE (preferred if executor judges the heuristic too fragile):** Split the try/catch inside `#processToken` around the match lookup vs replaceToken call. Since `replaceToken` internally does both import and creation, the executor may instead modify `replaceToken` to set a stage marker on thrown errors:
   ```javascript
   // In replaceToken, before import stage:
   // error.stage = "import";  // NOT recommended — mutating caught errors is fragile
   ```

   The simplest correct approach: In `#processToken`, wrap `replaceToken` in a single catch. Since ALL errors from `replaceToken` are either import or creation related, and the method is structured linearly (import first, then create), we can determine the stage by whether `worldActor` was successfully obtained. However since `replaceToken` is a single method, we don't have that visibility from `#processToken`.

   **FINAL recommended approach:** Use a simple heuristic based on error message content. The error messages in `replaceToken` are controlled by our code:
   - `"Failed to import actor"` → import_failed
   - `"Failed to create new token"` → creation_failed
   - Any `pack.getDocument` error → import_failed (fetching from compendium is pre-import)
   - Any `createEmbeddedDocuments` error → creation_failed
   - Any `deleteEmbeddedDocuments` error → creation_failed

   Default to "creation_failed" for unknown errors (import is attempted first, so if we got past it, creation is the likely failure point).

   static #reportResults(replaced, notFound, importFailed, creationFailed) {
     if (replaced > 0) {
       ui.notifications.info(game.i18n.format("NPC_REPLACER.Complete", { count: replaced }));
     }

     if (notFound.length > 0) {
       ui.notifications.warn(game.i18n.format("NPC_REPLACER.NotFoundCount", { count: notFound.length }));
       Logger.log("Creatures not found in compendiums:", notFound);
     }

     const totalErrors = importFailed.length + creationFailed.length;
     if (totalErrors > 0) {
       ui.notifications.error(game.i18n.format("NPC_REPLACER.SummaryPartialFailure", {
         replaced,
         noMatch: notFound.length,
         importFailed: importFailed.length,
         creationFailed: creationFailed.length
       }));
       if (importFailed.length > 0) Logger.log("Import failures:", importFailed);
       if (creationFailed.length > 0) Logger.log("Creation failures:", creationFailed);
     }

     Logger.log(`Replacement complete: ${replaced} replaced, ${notFound.length} not found, ${importFailed.length} import failures, ${creationFailed.length} creation failures`);
   }

### ERR-03 Tests (easier to test — public API)

```javascript
describe("ERR-03: Per-compendium load error tracking", () => {
  beforeEach(() => {
    CompendiumManager.clearCache();
    ui.notifications.error = vi.fn();
    game.i18n.format = vi.fn((key, data) => `${key}: ${JSON.stringify(data)}`);
  });

  it("getLastLoadErrors returns empty array when no errors occurred", () => {
    expect(CompendiumManager.getLastLoadErrors()).toEqual([]);
  });

  it("records per-pack errors during loadMonsterIndex", async () => {
    const failingPack = {
      collection: "dnd-broken.monsters",
      metadata: { packageName: "dnd-broken", label: "Broken Pack" },
      documentName: "Actor",
      getIndex: vi.fn().mockRejectedValue(new Error("Network timeout")),
    };

    // Setup: make getEnabledCompendiums return our failing pack
    const mockPacks = [failingPack];
    game.packs.filter = vi.fn(pred => mockPacks.filter(pred));
    game.settings.get = vi.fn().mockReturnValue('["all"]');

    await CompendiumManager.loadMonsterIndex(true);

    const errors = CompendiumManager.getLastLoadErrors();
    expect(errors).toHaveLength(1);
    expect(errors[0].packId).toBe("dnd-broken.monsters");
    expect(errors[0].packLabel).toBe("Broken Pack");
    expect(errors[0].error).toBe("Network timeout");
  });

  it("resets errors on each loadMonsterIndex call", async () => {
    // First call with failure
    const failingPack = {
      collection: "dnd-broken.monsters",
      metadata: { packageName: "dnd-broken", label: "Broken Pack" },
      documentName: "Actor",
      getIndex: vi.fn().mockRejectedValue(new Error("fail")),
    };
    game.packs.filter = vi.fn(pred => [failingPack].filter(pred));
    game.settings.get = vi.fn().mockReturnValue('["all"]');

    await CompendiumManager.loadMonsterIndex(true);
    expect(CompendiumManager.getLastLoadErrors()).toHaveLength(1);

    // Second call with success (no packs)
    CompendiumManager.clearCache();
    game.packs.filter = vi.fn(() => []);
    await CompendiumManager.loadMonsterIndex(true);
    expect(CompendiumManager.getLastLoadErrors()).toEqual([]);
  });

  it("clearCache resets lastLoadErrors", async () => {
    // Populate errors
    const failingPack = {
      collection: "dnd-broken.monsters",
      metadata: { packageName: "dnd-broken", label: "Broken Pack" },
      documentName: "Actor",
      getIndex: vi.fn().mockRejectedValue(new Error("fail")),
    };
    game.packs.filter = vi.fn(pred => [failingPack].filter(pred));
    game.settings.get = vi.fn().mockReturnValue('["all"]');

    await CompendiumManager.loadMonsterIndex(true);
    expect(CompendiumManager.getLastLoadErrors()).toHaveLength(1);

    CompendiumManager.clearCache();
    expect(CompendiumManager.getLastLoadErrors()).toEqual([]);
  });

  it("getDebugAPI exposes getLastLoadErrors", () => {
    const api = NPCTokenReplacerController.getDebugAPI();
    expect(typeof api.getLastLoadErrors).toBe("function");
  });

  it("getLastLoadErrors returns a copy, not the original array", async () => {
    const errors1 = CompendiumManager.getLastLoadErrors();
    const errors2 = CompendiumManager.getLastLoadErrors();
    expect(errors1).not.toBe(errors2);  // Different array references
    expect(errors1).toEqual(errors2);   // Same content
  });
});
```
